	" match scheduled for ", a.timelimit, " or ", a.fallrule

left join ringpro.tbl_wrestlers b
on a.wrestlerID_1 = b.wrestlerID
left join ringpro.tbl_wrestlers c
on a.wrestlerID_2 = c.wrestlerID
